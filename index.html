// Juego 1: Contar Objetos

function loadCountingGame() {

const content = document.getElementById('gameContent');

const count = Math.floor(Math.random() * (5 + level * 2)) + 3;

const emoji = emojis[Math.floor(Math.random() * emojis.length)];

currentProblem = { count, answer: count };

let html = '<div class="counting-area">';

html += '<div class="counting-container" id="countingContainer">';

for (let i = 0; i < count; i++) {

html += `<span class="count-item" style="animation-delay: ${i * 0.1}s">${emoji}</span>`;

}

html += '</div>';

html += '<div class="number-options" id="numberOptions"></div>';

html += '</div>';

content.innerHTML = html;

// Generar opciones

const options = [count];

while (options.length < 4) {

const num = Math.floor(Math.random() * (count + 5)) + 1;

if (!options.includes(num) && num > 0) options.push(num);

}

options.sort(() => Math.random() - 0.5);

const optionsContainer = document.getElementById('numberOptions');

options.forEach(num => {

const btn = document.createElement('button');

btn.className = 'number-btn';

btn.textContent = num;

btn.onclick = () => checkCountingAnswer(num, btn);

optionsContainer.appendChild(btn);

});

}

function checkCountingAnswer(answer, btn) {

if (answer === currentProblem.answer) {

btn.classList.add('correct');

showFeedback(true, '¡Muy bien! 🎉');

score += 10;

progress = Math.min(progress + 20, 100);

updateScore();

updateProgress();

setTimeout(() => {

if (progress >= 100) {

levelUp();

} else {

loadCountingGame();

}

}, 1500);

} else {

btn.classList.add('wrong');

showFeedback(false, 'Intenta de nuevo 💪');

setTimeout(() => btn.classList.remove('wrong'), 500);

}

}

html += '</div>';

html += '</div>';

html += '</div>';

html += '<div class="number-options" style="margin-top: 30px;" id="subtractionOptions"></div>';

html += '</div>';

const options = [total - eat];

while (options.length < 4) {

const num = Math.floor(Math.random() * total) + 1;

if (!options.includes(num) && num >= 0) options.push(num);

}

}

currentProblem.eaten++;

if (currentProblem.eaten === currentProblem.eat) {

document.getElementById('remainingDisplay').textContent = currentProblem.answer;

}

}

}

}

}

// Juego 4: Secuencias

function loadSequenceGame() {

const content = document.getElementById('gameContent');

const start = Math.floor(Math.random() * 10) + 1;

const step = Math.floor(Math.random() * 3) + 1;

const hiddenIndex = Math.floor(Math.random() * 4) + 1;

const sequence = [];

for (let i = 0; i < 5; i++) {

sequence.push(start + (i * step));

}

currentProblem = { sequence, hiddenIndex, answer: sequence[hiddenIndex] };

let html = '<div style="text-align: center;">';

html += '<h3 style="color: #666; margin-bottom: 20px;">¿Qué número sigue en la secuencia?</h3>';

html += '<div class="sequence-container" id="sequenceContainer">';

sequence.forEach((num, idx) => {

if (idx === hiddenIndex) {

html += `<div class="sequence-box empty" id="emptyBox" onclick="selectEmptyBox()">?</div>`;

} else {

html += `<div class="sequence-box filled">${num}</div>`;

}

});

html += '</div>';

// Opciones

const options = [sequence[hiddenIndex]];

while (options.length < 3) {

const num = sequence[hiddenIndex] + Math.floor(Math.random() * 10) - 5;

if (!options.includes(num) && num > 0) options.push(num);

}

options.sort(() => Math.random() - 0.5);

html += '<div class="sequence-options" id="sequenceOptions">';

options.forEach(num => {

html += `<div class="option-bubble" onclick="selectSequenceOption(${num}, this)">${num}</div>`;

});

html += '</div>';

html += '</div>';

content.innerHTML = html;

}

let selectedOption = null;

function selectSequenceOption(num, element) {

document.querySelectorAll('.option-bubble').forEach(el => el.classList.remove('selected'));

element.classList.add('selected');

selectedOption = num;

const emptyBox = document.getElementById('emptyBox');

emptyBox.textContent = num;

emptyBox.style.background = '#ffd700';

setTimeout(() => checkSequenceAnswer(num), 500);

}

function checkSequenceAnswer(answer) {

if (answer === currentProblem.answer) {

const emptyBox = document.getElementById('emptyBox');

emptyBox.style.background = '#4cd964';

emptyBox.style.color = 'white';

showFeedback(true, '¡Descubriste el patrón! 🔍✨');

score += 25;

progress = Math.min(progress + 25, 100);

updateScore();

updateProgress();

setTimeout(() => {

if (progress >= 100) {

levelUp();

} else {

loadSequenceGame();

}

}, 1500);

} else {

const emptyBox = document.getElementById('emptyBox');

emptyBox.style.background = '#ff6b6b';

emptyBox.style.color = 'white';

showFeedback(false, 'Observa bien los números 🔢');

setTimeout(() => {

emptyBox.textContent = '?';

emptyBox.style.background = '#f0f0f0';

emptyBox.style.color = 'inherit';

document.querySelectorAll('.option-bubble').forEach(el => el.classList.remove('selected'));

}, 1000);

}

}

// Sistema de puntuación y niveles

function updateScore() {

document.getElementById('score').textContent = score;

// Actualizar estrellas visuales

const stars = Math.floor(score / 50) + 3;

let starsText = '';

for (let i = 0; i < Math.min(stars, 5); i++) starsText += '🌟';

document.getElementById('starsCount').textContent = starsText || '🌟';

}

function levelUp() {

level++;

progress = 0;

updateProgress();

document.getElementById('level').textContent = level;

showFeedback(true, `¡Subiste al Nivel ${level}! 🚀🎊`);

createConfetti();

setTimeout(() => {

loadGame();

}, 2000);

}

// Feedback visual

function showFeedback(success, message) {

const overlay = document.getElementById('overlay');

const feedback = document.getElementById('feedback');

const icon = document.getElementById('feedbackIcon');

const text = document.getElementById('feedbackText');

icon.textContent = success ? '🎉' : '🤔';

text.textContent = message;

text.style.color = success ? '#4cd964' : '#ff6b6b';

overlay.classList.add('show');

feedback.classList.add('show');

if (success) createConfetti();

setTimeout(() => {

overlay.classList.remove('show');

feedback.classList.remove('show');

}, 1500);

}

// Efecto de confeti

function createConfetti() {

for (let i = 0; i < 30; i++) {

const confetti = document.createElement('div');

confetti.className = 'confetti';

confetti.style.left = Math.random() * 100 + '%';

confetti.style.background = colors[Math.floor(Math.random() * colors.length)];

confetti.style.animationDuration = (Math.random() * 2 + 2) + 's';

document.body.appendChild(confetti);

setTimeout(() => confetti.remove(), 4000);

}

}
